create or replace PROCEDURE SPRING_AUTHENTICATION 
(
  USERNAME IN VARCHAR2 
, PASSWORD IN VARCHAR2 
, RESULT OUT VARCHAR2 
) AS 
  CNT number(5);
BEGIN
   SELECT COUNT(*) INTO CNT FROM USERSLIST WHERE UNAME=USERNAME AND PWD=PASSWORD;
   if(CNT<>0) then
     RESULT:='VALID CREDENTIALS';
     else
     RESULT:='INVALID CREDENTIALS';
  END IF;     
  
END SPRING_AUTHENTICATION;
=============================

CREATE TABLE "SYSTEM"."USERSLIST" 
   (	"UNAME" VARCHAR2(20 BYTE), 
	"PWD" VARCHAR2(20 BYTE), 
	 PRIMARY KEY ("UNAME"))
